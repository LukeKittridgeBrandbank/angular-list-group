Select as with list-group.

(selectable=[multiple], selected-items, before-selection-change, after-selection-change)

Controller


$scope.colors=[ 'Red', 'Orange', 'Yellow', 'Green', 'Blue', 'Indigo', 'Violet' ];
$scope.logs = [];
$scope.selectedItems = [];
$scope.beforeSelectionChangeHandler = function(item) {
	$scope.logs.push('beforeSelectionChangeHandler(' + item + ')');
	return true;
};
$scope.afterSelectionChangeHandler = function(item) {
	$scope.logs.push('afterSelectionChangeHandler(' + item + ')');
};
$scope.$watchCollection('selectedItems',function (newVal, oldVal) {
	$scope.logs.push(angular.toJson(newVal));
});

Markup

<list-group-editor items="colors" selectable="multiple"
	selected-items="selectedItems"
	before-selection-change="beforeSelectionChangeHandler(item)"
	after-selection-change="afterSelectionChangeHandler(item)">
</list-group-editor>
