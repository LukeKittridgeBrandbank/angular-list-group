Use datasource attribute to populate items from a remote datasource.

datasource must be a function returning a promise.

Controller

	$scope.myDatasource = function() {
		var deferred = $q.defer();
		var promise = deferred.promise;
		$timeout(function() {
			deferred.resolve($scope.colors);
		}, 1000);
		return promise;
	}

Markup

	
<list-group datasource="myDatasource"></list-group>
