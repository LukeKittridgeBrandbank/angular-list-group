To enable the list filtering add a filterable attribute

Controller

$scope.colors=[ 'Red', 'Orange', 'Yellow', 
	'Green', 'Blue', 'Indigo', 'Violet' ];

Markup

<list-group items="colors" 
	selectable="true" filterable> 
</list-group>
Add a placeholder to the input field

Controller

$scope.colors=[ 'Red', 'Orange', 'Yellow', 
	'Green', 'Blue', 'Indigo', 'Violet' ];

$scope.filterable = {
	placeholder : 'Filter colors...'
};

Markup

<list-group items="colors" 
	selectable="true" filterable="filterable"> 
</list-group>
If you want to trigger filter yourself, give filterable auto property a false value

Controller

$scope.colors=[ 'Red', 'Orange', 'Yellow', 
	'Green', 'Blue', 'Indigo', 'Violet' ];

$scope.filterable02 = {
	auto : false
};
<list-group items="colors" 
	selectable="true" filterable="filterable02"> 
</list-group>
Override default comparator contains giving filterable comparator property value one of eq, neq, startsWith or endsWith

Controller

$scope.colors=[ 'Red', 'Orange', 'Yellow', 
	'Green', 'Blue', 'Indigo', 'Violet' ];

$scope.filterable03 = {
	comparator : 'startsWith'
};

Markup

<list-group items="colors" 
	filterable="filterable03"> 
</list-group>
If you are case sensitive set property ignoreCase to false on your filterable object

Controller

$scope.colors=[ 'Red', 'Orange', 'Yellow', 
	'Green', 'Blue', 'Indigo', 'Violet' ];

$scope.filterable04 = {
	comparator : 'startsWith',
	ignoreCase : false
};

Markup

<list-group items="colors" 
	filterable="filterable04"> 
</list-group>
