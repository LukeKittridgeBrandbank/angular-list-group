Use attribute contextual-class to add css class to items

Controller

$scope.colors=[ 'Red', 'Orange', 'Yellow', 
	'Green', 'Blue', 'Indigo', 'Violet' ];

Markup

<list-group items="colors"
	contextual-class="list-group-item-success"> 
</list-group>
Or if you want the class to be specified from the controller

Controller

$scope.colors=[ 'Red', 'Orange', 'Yellow', 
	'Green', 'Blue', 'Indigo', 'Violet' ];
$scope.myContextualClass='list-group-item-warning';

Markup

<list-group items="colors"
	contextual-class="{{myContextualClass}}">
</list-group>
Pass in a callback function to control which class is to be applied to each item

Controller

$scope.colors=[ 'Red', 'Orange', 'Yellow', 
	'Green', 'Blue', 'Indigo', 'Violet' ];

$scope.contextualClassFnHandler = function(item){
	var clazz = 'list-group-item-success';
	if (item == 'Orange') {
		clazz = 'list-group-item-warning';
	}
	return clazz; 
}

Markup

<list-group items="colors" 
	contextual-class="contextualClassFnHandler(item)"> 
</list-group>
