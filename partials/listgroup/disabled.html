Use attribute disabled to enable/disable items

Controller

$scope.colors=[ 'Red', 'Orange', 'Yellow', 
	'Green', 'Blue', 'Indigo', 'Violet' ];

$scope.disabled = true;

Markup

	
<list-group items="colors" selectable="true"
	disabled="disabled"> 
</list-group>
disabled
Pass in a callback function to control items to be disabled

Controller

$scope.colors=[ 'Red', 'Orange', 'Yellow', 
	'Green', 'Blue', 'Indigo', 'Violet' ];

$scope.disableFnHandler = function(item){
	return item != 'Orange'; 
}

Markup

<list-group items="colors" 
	selectable="true"
	disabled="disableFnHandler(item)"> 
</list-group>
