Add custom HTML using template attribute

Controller

$scope.paintings = [{
	'title' : 'Blue Dancers, c.1899',
	'paintedBy' : 'Edgar Degas',
	'location' : 'Pushkin Museum, Moscow, Russia',
	'year' : 1899,
	'dimensions' : '25.59 inch wide x 25.59 inch high',
	'url' : 'http://www.edgar-degas.org/Blue-Dancers,-c.1899.jpg'},
   ...
];

$scope.paintingTpl = '<div class="row painting-item"><div class="pull-left"><a href="#" class="thumbnail"><img ng-src="{{item.url}}"></img></a></div><div class="pull-left"><h4 class="list-group-item-heading">{{item.title}}</h4> <p class="list-group-item-text">{{item.paintedBy}}</p><p class="list-group-item-text">{{item.year}}</p></div></div>';

Markup

<list-group items="paintings"
	selectable="true" template="paintingTpl"></list-group>
Or load HTML from URL using template-url attribute 
$scope.paintings = [{
	'title' : 'Blue Dancers, c.1899',
	'paintedBy' : 'Edgar Degas',
	'location' : 'Pushkin Museum, Moscow, Russia',
	'year' : 1899,
	'dimensions' : '25.59 inch wide x 25.59 inch high',
	'url' : 'http://www.edgar-degas.org/Blue-Dancers,-c.1899.jpg'},
   ...
];

 $scope.templateUrl = 'partials/painting-list-item.tpl.html';

Markup

<list-group items="paintings"
		template-url="'partials/painting-list-item.tpl.html'"></list-group>

// or

<list-group items="paintings"
	template-url="templateUrl"></list-group>
